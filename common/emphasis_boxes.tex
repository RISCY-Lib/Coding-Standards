% Incorrect Style Box
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Create a incorrect style counter
\newcounter{badbox}[chapter]\setcounter{badbox}{0}
\renewcommand{\thebadbox}{\arabic{chapter}.\arabic{badbox}}
\newcommand{\badboxautorefname}{Incorrect}

% Create the definition box
\newenvironment{badbox}[1][]{
  \refstepcounter{badbox}

  % If no title given
  \ifstrempty{#1}
  {
    \mdfsetup{
      frametitle={
        \tikz[baseline=(current bounding box.east), outer sep=0pt]
        \node[anchor=east,rectangle,fill=red!30]
          %{\strut Incorrect~\thebadbox};
          {\strut Incorrect};
      }
    }
  }{
    \mdfsetup{
      frametitle={
        \tikz[baseline=(current bounding box.east), outer sep=0pt]
        \node[anchor=east,rectangle,fill=red!30]
          % {\strut Incorrect~\thebadbox:~#1};
          {\strut Incorrect:~#1};
      }
    }
  }

  % Regardless of title given
  \mdfsetup{
    innertopmargin=12pt,
    linecolor=red!75,
    linewidth=2pt,
    topline=true,
    frametitleaboveskip=\dimexpr-\ht\strutbox\relax,
    nobreak=true
  }

  \vspace{5pt}
  \begin{mdframed}[]\relax%
}{
  \end{mdframed}
}
